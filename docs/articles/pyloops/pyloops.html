Python Loops

Here are three ways to build a for loop in Python. 

#!/usr/bin/env python

def counter(_from,to,by):
    while _from<=to:
        yield _from
        _from += by

def loop1(_from,to,by=1):
    for c in range(_from,to+1,by):
        pass

def loop2(_from,to,by=1):
    while _from<=to:
        _from += by
        
def loop3(_from,to,by=1):
    for c in counter(_from,to,by):
        pass
